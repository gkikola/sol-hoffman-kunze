\section{The Algebra of Polynomials}

\Exercise1 Let $F$ be a subfield of the complex numbers and let $A$ be
the following $2\times2$ matrix over $F$,
\begin{equation*}
  A =
  \begin{bmatrix}
    2 & 1 \\
    -1 & 3
  \end{bmatrix}.
\end{equation*}
For each of the following polynomials $f$ over $F$, compute $f(A)$.
\begin{enumerate}
\item $f = x^2 - x + 2$
  \begin{solution}
    We have
    \begin{align*}
      f(A)
      &=
      \begin{bmatrix}
        2 & 1 \\
        -1 & 3
      \end{bmatrix}^2
      -
      \begin{bmatrix}
        2 & 1 \\
        -1 & 3
      \end{bmatrix}
      + 2
      \begin{bmatrix}
        1 & 0 \\
        0 & 1
      \end{bmatrix} \\
      &=
      \begin{bmatrix}
        3 & 5 \\
        -5 & 8
      \end{bmatrix}
      -
      \begin{bmatrix}
        2 & 1 \\
        -1 & 3
      \end{bmatrix}
      +
      \begin{bmatrix}
        2 & 0 \\
        0 & 2
      \end{bmatrix} \\
      &=
      \begin{bmatrix}
        3 & 4 \\
        -4 & 7
      \end{bmatrix}. \qedhere
    \end{align*}
  \end{solution}

\item $f = x^3 - 1$
  \begin{solution}
    Since
    \begin{equation*}
      A^3 =
      \begin{bmatrix}
        1 & 18 \\
        -18 & 19
      \end{bmatrix},
    \end{equation*}
    we get
    \begin{equation*}
      f(A) = A^3 - I =
      \begin{bmatrix}
        0 & 18 \\
        -18 & 18
      \end{bmatrix}. \qedhere
    \end{equation*}
  \end{solution}

\item $f = x^2 - 5x + 7$
  \begin{solution}
    In this case, we have
    \begin{align*}
      f(A)
      &=
      \begin{bmatrix}
        2 & 1 \\
        -1 & 3
      \end{bmatrix}^2
      - 5
      \begin{bmatrix}
        2 & 1 \\
        -1 & 3
      \end{bmatrix}
      + 7
      \begin{bmatrix}
        1 & 0 \\
        0 & 1
      \end{bmatrix} \\
      &=
      \begin{bmatrix}
        3 & 5 \\
        -5 & 8
      \end{bmatrix}
      +
      \begin{bmatrix}
        -10 & -5 \\
        5 & -15
      \end{bmatrix}
      +
      \begin{bmatrix}
        7 & 0 \\
        0 & 7
      \end{bmatrix} \\
      &=
      \begin{bmatrix}
        0 & 0 \\
        0 & 0
      \end{bmatrix}. \qedhere
    \end{align*}
  \end{solution}
\end{enumerate}
